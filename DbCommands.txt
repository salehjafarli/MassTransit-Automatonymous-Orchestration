For Api2(Postgres) :

create table Companies(
CompanyId int not null generated always as identity,
CompanyName varchar(30) not null,
Primary key(CompanyId)
);
create Table Categories(
  CategoryId int not null generated always as identity,
  CategoryName varchar(30) not null,
  Primary key(CategoryId)
);
create Table Products
(
  ProductId int not null generated always as identity,
  ProductName varchar(30) not null,
  ProductCost DOUBLE PRECISION not null,
  CategoryId int not null,
  Primary key(ProductId),
  Constraint Fk_CategoryId Foreign key(CategoryId) references Categories(CategoryId)
);
create table Warehouses(
  WarehouseId int not null generated always as identity,
  WarehouseName varchar(30) not null,
  WarehouseAddress varchar(60) not null,
  CompanyId int not null,
  Primary key(WarehouseId),
  Constraint Fk_CompanyId Foreign key(CompanyId) references Companies(CompanyId)
);
create table WarehouseData(
  WarehouseId int not null,
  ProductId int not null,
  PRIMARY KEY (WarehouseId,ProductId),
  Constraint Fk_WarehouseId Foreign key(WarehouseId) references Warehouses(WarehouseId),
  Constraint Fk_ProductId Foreign key(ProductId) references Products(ProductId)
);
